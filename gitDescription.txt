move "fileName.txt" "gitProject"	moves the file to the directory
cd gitProject		Goes to the projectfolder
git init		Creates an empty git reposotory
ls -a			Shows ALL the items, also the hidden
git status		Show the status

git add --all		Adds all changes, files and directories to the commit
git commit -m "The first commit message"	A git commit with a message
